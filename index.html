    <script>

        const imgElement = document.getElementById('catImg');

        const loaderText = document.querySelector('.loader');


        // URL публичного API (TheCatAPI)

        const apiUrl = 'https://api.thecatapi.com/v1/images/search';


        async function fetchNewCat() {

            // 1. Скрываем старое фото и показываем текст загрузки

            imgElement.classList.remove('loaded');

            loaderText.style.display = 'block';

            

            try {

                // 2. Делаем запрос к API

                const response = await fetch(apiUrl);

                const data = await response.json();

                

                // API возвращает массив, берем первый элемент

                const catUrl = data[0].url;


                // 3. Устанавливаем новый источник изображения

                // Используем событие onload, чтобы показать фото только когда оно скачалось

                imgElement.onload = () => {

                    imgElement.classList.add('loaded');

                    loaderText.style.display = 'none';

                };

                

                imgElement.src = catUrl;


            } catch (error) {

                console.error('Ошибка мяу-сервера:', error);

                loaderText.innerText = "Ошибка сети. Попробуйте снова!";

            }

        }


        // Загрузить первого кота при открытии страницы

        fetchNewCat();

    </script>
